254 size_t HHScanner::scanProcesses(HHScanReport &my_report)
255 {
256  size_t count = 0;
257  size_t scanned_count = 0;
258  size_t ignored_count = 0;
259 
260  HANDLE hProcessSnapShot = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
261  if (hProcessSnapShot == INVALID_HANDLE_VALUE) {
262  const DWORD err = GetLastError();
263  std::cerr << "[-] Could not create modules snapshot. Error: " << std::dec << err << std::endl;
264  return 0;
265  }
266 
267  PROCESSENTRY32 pe32 = { 0 };
268  pe32.dwSize = sizeof(PROCESSENTRY32);
269 
270  //check all modules in the process, including the main module:
271  if (!Process32First(hProcessSnapShot, &pe32)) {
272  CloseHandle(hProcessSnapShot);
273  std::cerr << "[-] Could not enumerate processes. Error: " << GetLastError() << std::endl;
274  return 0;
275  }
276  do {
277  // scan callback
278  const t_single_scan_status stat = scanNextProcess(pe32.th32ProcessID, pe32.szExeFile, my_report);
279  if (stat == SSCAN_IGNORED) ignored_count++;
280  if (stat == SSCAN_SUCCESS) scanned_count++;
281  count++;
282 
283  } while (Process32Next(hProcessSnapShot, &pe32));
284 
285  //close the handles
286  CloseHandle(hProcessSnapShot);
287 
288  printScanRoundStats(scanned_count, ignored_count);
289  return count;
