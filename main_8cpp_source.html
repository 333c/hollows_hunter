119 {
120  const int hdr_color = HEADER_COLOR;
121  const int param_color = HILIGHTED_COLOR;
122  const int separator_color = SEPARATOR_COLOR;
123 
124  const bool has_filter = filter.length() > 0 ? true : false;
125  bool has_any = false;
126 
127  std::map<std::string, void(*)(int)>::iterator itr;
128  for (itr = params_block.begin(); itr != params_block.end();itr++) {
129  const std::string &param = itr->first;
130  if (has_filter) {
131  stringsim_type sim_type = is_string_similar(param, filter);
132  if (sim_type != SIM_NONE) has_any = true;
133  }
134  else {
135  has_any = true;
136  }
137  }
138  if (!has_any) return 0;
139 
140  int p_color = param_color;
141  if (block_name.length()) {
142  print_in_color(separator_color, "\n---" + block_name + "---\n");
143  }
144  size_t counter = 0;
145  for (itr = params_block.begin(); itr != params_block.end();itr++) {
146  const std::string &param = itr->first;
147  if (filter.length() > 0) {
148  const stringsim_type sim_type = is_string_similar(param, filter);

150  if (sim_type == SIM_NONE) continue;

152  void(*info)(int) = itr->second;
153  if (!info) continue;
154  info(p_color);
155  counter++;
156  }
157  if (has_filter) {
