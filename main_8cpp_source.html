118size_t print_params_block(std::string block_name, std::map<std::string, void(*)(int)> params_block, const std::string &filter)
119{
120 const int hdr_color = HEADER_COLOR;
121 const int param_color = HILIGHTED_COLOR;
122 const int separator_color = SEPARATOR_COLOR;
123
124 const bool has_filter = filter.length() > 0 ? true : false;
125 bool has_any = false;
126
127 std::map<std::string, void(*)(int)>::iterator itr;
128 for (itr = params_block.begin(); itr != params_block.end();itr++) {
129 const std::string &param = itr->first;
130 if (has_filter) {
131 stringsim_type sim_type = is_string_similar(param, filter);
132 if (sim_type != SIM_NONE) has_any = true;
133 }
134 else {
135 has_any = true;
136 }
137 }
138 if (!has_any) return 0;
139
140 int p_color = param_color;
141 if (block_name.length()) {
142 print_in_color(separator_color, "\n---" + block_name + "---\n");
143 }
144 size_t counter = 0;
145 for (itr = params_block.begin(); itr != params_block.end();itr++) {
146 const std::string &param = itr->first;
147 if (filter.length() > 0) {
148 const stringsim_type sim_type = is_string_similar(param, filter);
149 p_color = (sim_type != SIM_NONE) ? ERROR_COLOR : param_color;
150 if (sim_type == SIM_NONE) continue;
151 }
152 void(*info)(int) = itr->second;
153 if (!info) continue;
154 info(p_color);
155 counter++;
156 }
157 if (has_filter) {
158 print_in_color(INACTIVE_COLOR, "\n[...]\n");
159 }
160 return counter;
161}

235std::string version_to_str(DWORD version)
236{
237 BYTE *chunks = (BYTE*) &version;
238 std::stringstream stream;
239 stream << std::hex <<
240 (int)chunks[3] << "." <<
241 (int)chunks[2] << "." <<
242 (int)chunks[1] << "." <<
243 (int)chunks[0];
244
245 return stream.str();
246}

371void deploy_scan(t_hh_params &hh_args)
372{
373 if (hh_args.pesieve_args.data >= pesieve::PE_DATA_SCAN_INACCESSIBLE && hh_args.pesieve_args.make_reflection == false) {
374 print_in_color(RED, "[WARNING] Scanning of inaccessible pages is enabled only in the reflection mode!\n");
375 }
376 hhunter::util::set_debug_privilege();
377 HHScanner hhunter(hh_args);
378 do {
379 HHScanReport *report = hhunter.scan();
380 if (report) {
381 hhunter.summarizeScan(report);
382 delete report;
383 }
384 if (!HHScanner::isScannerCompatibile()) {
385 compatibility_alert();
386 }
387 } while (hh_args.loop_scanning);
388}
